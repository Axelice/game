          const runRound = function({board}) {
            let steps = 0;
            let solution;
            while (steps < 5) {
              solution = {
                posX: Math.floor(Math.random() * 10),
                posY: Math.floor(Math.random() * 10)
              };
          
              if (board[solution.posX][solution.posY] === undefined) {
                steps = 5;
              } else {
                steps++;
              }
            }
            return solution;
          };
